88 - (void)keyboardWillShow:(NSNotification*)notification {
89  _keyboardRect = [[[notification userInfo] objectForKey:_UIKeyboardFrameEndUserInfoKey] CGRectValue];
90  _keyboardVisible = YES;
91 
92  UIView *firstResponder = [self findFirstResponderBeneathView:self];
93  if ( !firstResponder ) {
94  // No child view is the first responder - nothing to do here
95  return;
96  }
97 
98  if (!_priorInsetSaved) {
99  _priorInset = self.contentInset;
100  _priorInsetSaved = YES;
101  }
102 
103  // Shrink view's inset by the keyboard's height, and scroll to show the text field/view being edited
104  [UIView beginAnimations:nil context:NULL];
105  [UIView setAnimationCurve:[[[notification userInfo] objectForKey:UIKeyboardAnimationCurveUserInfoKey] intValue]];
106  [UIView setAnimationDuration:[[[notification userInfo] objectForKey:UIKeyboardAnimationDurationUserInfoKey] floatValue]];
107 
108  self.contentInset = [self contentInsetForKeyboard];
109  [self setContentOffset:CGPointMake(self.contentOffset.x,
110  [self idealOffsetForView:firstResponder withSpace:[self keyboardRect].origin.y - self.bounds.origin.y])
111  animated:YES];
112  [self setScrollIndicatorInsets:self.contentInset];
113 
114  [UIView commitAnimations];
115 }

183 - (UIView*)findFirstResponderBeneathView:(UIView*)view {
184  // Search recursively for first responder
185  for ( UIView *childView in view.subviews ) {
186  if ( [childView respondsToSelector:@selector(isFirstResponder)] && [childView isFirstResponder] ) return childView;
187  UIView *result = [self findFirstResponderBeneathView:childView];
188  if ( result ) return result;
189  }
190  return nil;
191 }

193 - (void)findTextFieldAfterTextField:(UIView*)priorTextField beneathView:(UIView*)view minY:(CGFloat*)minY foundView:(UIView**)foundView {
194  // Search recursively for text field or text view below priorTextField
195  CGFloat priorFieldOffset = CGRectGetMinY([self convertRect:priorTextField.frame fromView:priorTextField.superview]);
196  for ( UIView *childView in view.subviews ) {
197  if ( childView.hidden ) continue;
198  if ( ([childView isKindOfClass:[UITextField class]] || [childView isKindOfClass:[UITextView class]]) ) {
199  CGRect frame = [self convertRect:childView.frame fromView:view];
200  if ( childView != priorTextField && CGRectGetMinY(frame) >= priorFieldOffset && CGRectGetMinY(frame) < *minY ) {
201  *minY = CGRectGetMinY(frame);
202  *foundView = childView;
203  }
204  } else {
205  [self findTextFieldAfterTextField:priorTextField beneathView:childView minY:minY foundView:foundView];
206  }
207  }
208 }

227 -(CGFloat)idealOffsetForView:(UIView *)view withSpace:(CGFloat)space {
228 
229  // Convert the rect to get the view's distance from the top of the scrollView.
230  CGRect rect = [view convertRect:view.bounds toView:self];
231 
232  // Set starting offset to that point
233  CGFloat offset = rect.origin.y;
234 
235 
236  if ( self.contentSize.height - offset < space ) {
237  // Scroll to the bottom
238  offset = self.contentSize.height - space;
239  } else {
240  if ( view.bounds.size.height < space ) {
241  // Center vertically if there's room
242  offset -= floor((space-view.bounds.size.height)/2.0);
243  }
244  if ( offset + space > self.contentSize.height ) {
245  // Clamp to content size
246  offset = self.contentSize.height - space;
247  }
248  }
249 
250  if (offset < 0) offset = 0;
251 
252  return offset;
253 }
