21 /** Eigene IP Adresse herausfinden.
22  * Evtl. muss en1 auf en0 gestellt werden.
23  */
24  -(NSString*) getOwnIpAddress {
25 
26  NSString *address = @"error";
27  struct ifaddrs *interfaces = NULL;
28  struct ifaddrs *temp_addr = NULL;
29  int success = 0;
30  // retrieve the current interfaces - returns 0 on success
31  success = getifaddrs(&interfaces);
32  if (success == 0) {
33  // Loop through linked list of interfaces
34  temp_addr = interfaces;
35  while(temp_addr != NULL) {
36  if(temp_addr->ifa_addr->sa_family == AF_INET) {
37  // Check if interface is en0 which is the wifi connection on the iPhone
38  if([[NSString stringWithUTF8String:temp_addr->ifa_name] isEqualToString:@"en0"] || [[NSString stringWithUTF8String:temp_addr->ifa_name] isEqualToString:@"en1"]) {
39  // Get NSString from C String
40  address = [NSString stringWithUTF8String:inet_ntoa(((struct sockaddr_in *)temp_addr->ifa_addr)->sin_addr)];
41 
42  }
43 
44  }
45 
46  temp_addr = temp_addr->ifa_next;
47  }
48  }
49  // Free memory
50  freeifaddrs(interfaces);
51  return address;
52 
53 }

256 -(void) deleteOldDevices {
257 
258  int timeDeleteDevice = 10; // Nach wie vielen Sekunden soll ein Gerät als offline angenommen werden?
259 
260  int time_current = [[self getCurrentTimeStamp] intValue];
261 
262  for(int i=0; i<[devices count]; i++) {
263  int time_last_seen = [[[devices objectAtIndex:i] objectForKey:@"last_seen"] intValue];

294  * Geräteinformationen abspeichern und evtl. Antwortbroadcast absetzen.
295  */
296 -(void)pushUp:(void*)data lenght:(unsigned int) len {
297 
298  NSLog(@"[Devices] Empfange Broadcast ...");
299 
300  // Empfangende Daten (char) in NSString umwandeln
301  NSString *data_string = [NSString stringWithFormat:@"%s",data];
302  //NSLog(@"RECSTR: %s",data);
303 
304  // Informationen aus empfangenen Daten auslesen
305  NSArray *devinfo = [data_string componentsSeparatedByString:@";;"];
306  //NSLog(@"REC DATA: %@",devinfo);
307 
308  // Prüfen, ob empfangene Daten das richtige Format haben
309  bool bad_format = false;
310  bool ext_format = false;
311  if ([devinfo count] != 7 && [devinfo count] != 8) {
312  NSLog(@"[Devices] pushUp: Empfangene Daten haben falsches Format: %@",devinfo);
313  bad_format = true;
314  if ([devinfo count] == 8) { ext_format = true; }
315  }
316 
317  // Prüfen, ob dieses Gerät schon in devices Array vorhanden ist
318  bool known_device=false;
319 
320 
321 
322  for(int i=0; i<[devices count]; i++) {
323  if([[[devices objectAtIndex:i] objectForKey:@"id"] isEqual: [devinfo objectAtIndex:0]]) {
324  known_device = true;
325 
326  //NSLog(@"Dev1: %@ enthält Dev2: %@",[[devices objectAtIndex:i] objectForKey:@"id"], [devinfo objectAtIndex:0]);
327 
328  // Last-Seen Time von bekanntem Gerät updaten
329  [[devices objectAtIndex:i] setObject:[self getCurrentTimeStamp] forKey:@"last_seen"];
330 
331  //NSLog(@"DEVINFO: %@",devinfo);
332 
333  // Usernamen updaten
334  if([[devinfo objectAtIndex:0] isEqual:[self getOwnId]]) { // Eigenen Username updaten
335  [[devices objectAtIndex:i] setObject:[self getCurrentUsernameFromUI] forKey:@"username"];
336  } else { // Fremden Usernamen updaten
337  [[devices objectAtIndex:i] setObject:[devinfo objectAtIndex:1] forKey:@"username"];
338  }
339 
340  //NSLog(@"[Devices] Aktive Geräte: %@",devices);
341 
342  }
343  }
344  // Falls Gerät noch nicht bekannt ist und das Format stimmt, in Array "devices" speichern
345  if(!known_device && !bad_format) {
346 
347  // devinfo in Dictionary umwandeln und in Array devices speichern
348  NSMutableDictionary *dic_devinfo = [[NSMutableDictionary alloc] initWithObjectsAndKeys: [devinfo objectAtIndex:0], @"id", [devinfo objectAtIndex:1], @"username", [devinfo objectAtIndex:2], @"ip", [devinfo objectAtIndex:3], @"publickey1", [devinfo objectAtIndex:4], @"publickey2", [devinfo objectAtIndex:5], @"last_seen", nil];
349 
350  [devices addObject:dic_devinfo];
351 
352  // Falls gewünscht, auf Broadcast mit eigenen Geräteinformationen antworten
353  if(!ext_format && [[devinfo objectAtIndex:6] isEqual:@"1"]) {
354  //[self answerBroadcast];
355  } else if (ext_format && [[devinfo objectAtIndex:7] isEqual:@"1"]){ // Workaround, falls iPad Simualtor wieder einen zusätzlichen Paramter empfängt.
356  //[self answerBroadcast];
357  }
358  }
359 
360  // Aktive Userliste an Userinterface übergeben
361  NSArray *activeUsers = [self listActiveUsernames];
362  if([activeUsers count] > 0) {
363  //NSLog(@"[Devices] Announcing ActiveUsers to Userinterface: %@",activeUsers);
