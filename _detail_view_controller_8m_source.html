116 /* Zeigt die Nachricht des Senders an.
117  *
118  * @param sender Der Absender der Nachricht
119  * @param message Die anzuzeigende Nachricht
120  */
121 - (void) displayMessage:(NSString*)message From:(NSString*)sender {
122  NSDate *currentTime = [NSDate date];
123  NSDateFormatter *dateFormatter = [[NSDateFormatter alloc] init];
124  [dateFormatter setDateFormat:@"yyyy-MM-dd HH:mm:ss'\n'"];
125  NSString *timeFormated = [dateFormatter stringFromDate: currentTime];
126  NSMutableAttributedString *chatText = [self.display.attributedText mutableCopy];
127 
128  UIColor *color = [UIColor redColor];
129  NSMutableParagraphStyle *paragrahStyle = [[NSMutableParagraphStyle alloc] init];
130  [paragrahStyle setAlignment:NSTextAlignmentLeft];
131  if([sender isEqualToString:self.userName]) {
132  color = [UIColor blueColor];
133  [paragrahStyle setAlignment:NSTextAlignmentRight];
134  }
135 
136  NSAttributedString *senderAttributed = [[NSAttributedString alloc] initWithString: [sender stringByAppendingString:@": "] attributes:@{
137  NSFontAttributeName: [UIFont fontWithName:@"HelveticaNeue-Bold" size:16.0],
138  NSForegroundColorAttributeName: color,
139  NSParagraphStyleAttributeName: paragrahStyle}];
140  NSAttributedString *timeAttributed = [[NSAttributedString alloc] initWithString: timeFormated attributes:@{
141  NSFontAttributeName: [UIFont fontWithName:@"HelveticaNeue" size:12.0],
142  NSForegroundColorAttributeName: [UIColor blackColor],
143  NSParagraphStyleAttributeName: paragrahStyle}];
144  NSAttributedString *messageAttributed = [[NSAttributedString alloc] initWithString: message attributes:@{
145  NSFontAttributeName: [UIFont fontWithName:@"HelveticaNeue" size:16.0],
146  NSForegroundColorAttributeName: color,
147  NSParagraphStyleAttributeName: paragrahStyle}];
148  NSAttributedString *newLineAttributed = [[NSAttributedString alloc] initWithString: @"\n\n"];
149  [chatText appendAttributedString:timeAttributed];
150  [chatText appendAttributedString:senderAttributed];
151  [chatText appendAttributedString:messageAttributed];
152  [chatText appendAttributedString:newLineAttributed];
153 
154  [self.display setAttributedText:chatText];
155  [self.display scrollRangeToVisible:NSMakeRange([self.display.text length], 0)];
156 }

190 - (void)alertView:(UIAlertView *)alertView clickedButtonAtIndex:(NSInteger)buttonIndex
191 {
192  UITextField *textField = [alertView textFieldAtIndex:0];
193  if (buttonIndex == 0) {
194  if ([textField.text isEqualToString:@""]) {
195  self.userName = nil;
196  } else {
197  self.userName = textField.text;
198  }
199  NSLog(@"User name input: %@",textField.text);
200  }
201  if (buttonIndex == 1) {
202  self.userName = nil;
203  NSLog(@"User name input cancelled");
204  }
205  NSUserDefaults *defaults = [NSUserDefaults standardUserDefaults];
206  [defaults setObject:self.userName forKey:@"userName"];
207  [defaults synchronize];
208 }
