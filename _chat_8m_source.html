16 /** Block responsible for sending and receiving messages using a XML based protocol.
17  * ## Protocol
18  * The communication works so that when A needs to send a message to B, A will connect
19  * to B send the message and close the connection. This way only one way communication
20  * is being used in the TCP stream.
21  * All messages are sent packed in the a XML based protocol defined in:
22  * > NSString const *protocol
23  * To parse the protocol, the RE pattern is being used:
24  * > NSString const *pattern
25  * The protocol also support private messages, which are not being used at the current
26  * implementation. The <i>from</i> and <i>to</i> fields are nicknames and the ones starting with "@"
27  * are reserved, since "@All" means the message should be sent to everyone.
28  *
29  * ## Transmited data
30  * The content that is transmited is always the message packed in the protocol and this
31  * is encrypted with the public key of the user who is receiving the connection. The
32  * encryption is done with the Class Verschluesselung and the publick key can be
33  * retrieved from the Devices.
34  *
35  * ## Sendinng messages
36  * When sending a message, a TCP connection is stablished to all users and encrypted
37  * with its respectively public key.
38  *
39  * ## Server
40  * When initiated, a TCP server is started. It will received connections from other users
41  * and forward the received data to recvMessage:data.
42  */
43 @implementation Chat

47 /** Message XML based protocol. */
48 NSString const *protocol=@"<?xml version=\"1.0\" encoding=\"UTF-8\" >\n <message>\n <from>%@</from>\n <to>@All</to>\n <body>%@</body>\n </message>\n</xml>";

50 /** Pattern for RE that parses the protocol. */
51 NSString const *pattern=@"<?xml version=\"1.0\" encoding=\"UTF-8\" >\n <message>\n <from>(.*)</from>\n <to>(.*)</to>\n <body>(.*)</body>\n </message>\n</xml>";

71 /** Receives a message from the underlying TCP server.
72  * This method should be called whenever a message was received by the listening TcpOperation.
73  * After parsing of the protocol, the message will be forwarded to the detailViewController
74  * for displaying.
75  *
76  * @param data Received message, encrypted with our public key.
77  */
78 - (void) recvMessage:(NSData*)data {
79  NSString *message = [app.verschluesselung decryptMessage:data];
80  NSRange range; range.location = 0; range.length=[message length];
81  NSTextCheckingResult *result = [re firstMatchInString:message options:0 range:range];
82  NSString *from = [message substringWithRange:[result rangeAtIndex:1]];
83  if ([from isEqualToString:[app.detailViewController getUserName]])
84  return;
85  message = [message substringWithRange:[result rangeAtIndex:3]];
86  [app.detailViewController displayMessage:message From:from];
87 }

89 /** Sends a message to all users.
90  * To send the message, a TCP connection will be opened to every user with TcpOperation.
91  * The message is sent encrypted with the user's public key.
92  *
93  * @param message The text to be sent.
94  */
95 - (void) sendMessage:(NSString*)message {
96  DetailViewController *dvc = app.detailViewController;
97  message=[[NSString alloc] initWithFormat:(NSString*)protocol, [dvc getUserName], message];
98  Devices *devices = [app devices];
99  __autoreleasing NSMutableArray *connections = [[NSMutableArray alloc] initWithCapacity:10]; // will store the TCP connections
100 
101  for (NSDictionary *device in [devices getDevices]) { //send to all users
102  if ([[device objectForKey:@"username"] isEqualToString:[app.detailViewController getUserName]])
103  continue;
104  printf(" --==-- Sending message to %s\n", [[device objectForKey:@"ip"] UTF8String]);
105  TcpOperation *tcp = [[TcpOperation alloc] initWith:self host:[device objectForKey:@"ip"] port:@"21568"];
106  [tcp performSelectorInBackground:@selector(start) withObject:nil];
107  [tcp sendPacket:[app.verschluesselung cryptMessage:message UserID:[device objectForKey:@"id"]]];
108  [connections addObject:tcp];
109  }
110 }
