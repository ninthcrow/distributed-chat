19 /** Send or reiceive UDP packets.
20  * The Block can be initiated as sender or receiver. The whole implementations
21  * was achieved using socket BSD system calls and performSelectorInBackground:
22  * for the background reception of packets.
23  *
24  * ## Sender
25  * In this mode the block will be sending the UDP packets given to the method
26  * writeToSocket: length: to the broadcast address. PORT is the destination
27  * port of the udp packets.
28  *
29  * ## Receiver
30  * When setup as receiver, self will bind to PORT and received data will be
31  * forwarded to upper Block using pushUp: length:.
32  *
33  * ## Error report
34  * Errors will be reported to using:
35  * > [[self top] reportError:@"some error"];
36  * Make sure to present this to the users.
37  *
38  * Data flow works just like any other Block, except that pushUp is called
39  * when something is received and pushDown sends the data. Sender UDP Block
40  * will not call pushUp and receiver UDP Blocks will not react to pushDown.
41  */

82 /** Setup the UDP socket.
83  * This is the core of the class and here is where the whole network
84  * socket setup and use takes place.
85  *
86  *This method should not be called from outside the class.
87  *
88  * It was implemented in a way that whenever we should receive or
89  * send, this same function must be called.
90  *
91  * @note
92  * This implementation follows the
93  * standards for network programming shown in
94  * [Beej's Guide to Network Programming](http://beej.us/guide/bgnet/output/html/multipage/index.html)
95  * , so please referete to it if something is not clear.
96  */
97 -(void)connect {
98  struct addrinfo hints, *res, *p;
99  int status;
100 
101  memset(&hints, 0, sizeof hints);
102  hints.ai_family=AF_UNSPEC;
103  hints.ai_socktype=SOCK_DGRAM;
104  if (!sender)
105  hints.ai_flags=AI_PASSIVE;
106 
107 
108  // [Markus] INFO: F체r Betrieb im WLAN zuhause oder ander UNI n채chste Zeile auskommentieren und 체bern채chste einkommentieren ;)
109  status=getaddrinfo(sender?broadcast:NULL, PORT, &hints, &res);
110  //status=getaddrinfo(sender?"255.255.255.255":NULL, PORT, &hints, &res);
111  if (status) {
112  [[self top] reportError:@"UDP.connect: DNS/address Error."];
113  return; }
114 
115  for (p=res;p!=NULL;p=p->ai_next) {
116  sockfd=socket(p->ai_family,p->ai_socktype,p->ai_protocol);
117  if (sockfd<0) continue;
118 
119  if (sender) status=connect(sockfd, p->ai_addr, p->ai_addrlen);
120  else status=bind(sockfd, p->ai_addr, p->ai_addrlen);
121  if (status) {close(sockfd) ; continue; }
122 
123  break; }
124 
125  freeaddrinfo(res);
126 
127  if (!p) {
128  [[self top] reportError:@"UDP.connect: Connection Error."];
129  return; }
130 
131  if (sender) {
132  int yes=0xFFFFFFFF;
133  if (setsockopt(sockfd, SOL_SOCKET, SO_BROADCAST, (const void*)&yes, sizeof(yes))) {
134  [[self top] reportError:@"UDP.connect: Failed to activate broadcast."];
135  return;
136  }
137  } else {
138  [self performSelectorInBackground:@selector(readFromSocket) withObject:nil];
139  }
140 }

142 /** Receive UDP packets.
143  * This method should not be called from outside the class.
144  *
145  * It is only called on receivers. It forward the received pakcet
146  *
147  * @note
148  * See note in connect.
149  * @see connect
150  */
151 -(void)readFromSocket {
152  @autoreleasepool{
153  [[NSThread currentThread] setName:@"UDP.recv"]; }
154  char msg[16*1024];
155  struct timeval timeout;
156  int status;
157  fd_set master, read_fds;
158 
159  timeout.tv_sec=0;
160  timeout.tv_usec=100000;
161 
162  FD_ZERO(&master);
163  FD_SET(sockfd, &master);
164 
165  [running lock];
166 
167  while (!stop) {
168  read_fds=master;
169  select(sockfd+1, &read_fds, NULL, NULL, &timeout);
170  if (FD_ISSET(sockfd, &read_fds)) {
171  status=recv(sockfd, msg, sizeof(msg), 0);
172  if (status<0) {
173  [[self top] reportError:@"UDP.readFromSocket: Receive error."];
174  break; }
175  [self pushUp:msg lenght:status]; } }
176  [running unlock];
177 }

183  close(sockfd);
184  NSLog(@"UDP.end: The socket was closed.");

186 }
